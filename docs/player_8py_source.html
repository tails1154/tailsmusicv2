68def menu_nav(event, selected, options, allowInterrupt: bool = False):
69 """
70 Handle menu navigation: back cycles BACK, skip cycles FORWARD, play/pause (okbutton/okbutton2) selects.
71 Returns tuple (selected, selected_action) where selected_action is True if play/pause was pressed.
72 """
73 key_event = categorize(event)
74 if key_event.keystate == 1:
75 key = key_event.keycode
76 if key == config['backbutton']:
77 selected = (selected - 1) % len(options)
78 if allowInterrupt:
79 speak_allowinter(options[selected])
80 else:
81 speak(options[selected])
82 elif key == config['skipbutton']:
83 selected = (selected + 1) % len(options)
84 if allowInterrupt:
85 speak_allowinter(options[selected])
86 else:
87 speak(options[selected])
88 elif key in [config['okbutton'], config['okbutton2']]:
89 return selected, True
90 return selected, False

93class CommandQueue:
94 def __init__(self, maxsize: int = 100, verbose: bool = True,
95 enable_multiprocessing: bool = False):
96 """
97 Initialize the command queue system with multiprocessing support.
98
99 Args:
100 maxsize: Maximum number of queued commands
101 verbose: Whether to print execution details
102 enable_multiprocessing: Enable inter-process communication
103 """
104 self.enable_multiprocessing = enable_multiprocessing
105
106 if enable_multiprocessing:
107 # line 67
108 self.manager = multiprocessing.Manager()
109 self.queue = self.manager.Queue(maxsize=maxsize) #line 69, nice
110 self.lock = self.manager.Lock()
111 else:
112 self.queue = queue.Queue(maxsize=maxsize)
113 self.lock = threading.Lock()
114
115 self.running = False
116 self.verbose = verbose
117 self.listener_thread: Optional[threading.Thread] = None
118 self.process: Optional[multiprocessing.Process] = None
119
120 def start_command_listener(self) -> None:
121 """Start the background command listener thread."""
122 if self.running:
123 return
124
125 self.running = True
126 self.listener_thread = threading.Thread(
127 target=self._command_listener,
128 daemon=True,
129 name="CommandListener"
130 )
131 self.listener_thread.start()
132
133 if self.verbose:
134 print("Command queue listener started")
135
136 def start_remote_processor(self) -> None:
137 """Start a separate process to handle command execution."""
138 if not self.enable_multiprocessing:
139 raise RuntimeError("Multiprocessing not enabled in constructor")
140
141 self.process = multiprocessing.Process(
142 target=self._remote_command_processor,
143 daemon=True
144 )
145 self.process.start()
146
147 if self.verbose:
148 print("Remote command processor started")
149
150 def _remote_command_processor(self) -> None:
151 """Process commands in a separate process."""
152 while True:
153 try:
154 cmd = self.queue.get(timeout=1)
155 self._execute_command(cmd)
156 except queue.Empty:
157 if not self.running:
158 break
159 except Exception as e:
160 print(f"Remote processor error: {e}")
161
162 def stop(self) -> None:
163 """Stop all components of the command queue."""
164 with self.lock:
165 self.running = False
166 if self.listener_thread:
167 self.listener_thread.join(timeout=1.0)
168 if self.process:
169 self.process.join(timeout=1.0)
170 if self.enable_multiprocessing:
171 self.manager.shutdown()
172
173 def _command_listener(self) -> None:
174 """Background thread that listens for console input."""
175 # note to self: what is the point of this??
176 while self.running:
177 try:
178 cmd = input(">>> " if sys.stdin.isatty() else "").strip()
179 if cmd:
180 self.put_command(cmd)
181 except (EOFError, KeyboardInterrupt):
182 break
183 except Exception as e:
184 print(f"Command input error: {e}")
185
186 def process_Command(self, timeout: float = 0.01) -> None:
187 """
188 Process pending commands in the queue (in current process).
189
190 Args:
191 timeout: Maximum time to wait for a command (seconds)
192 """
193 while True:
194 try:
195 cmd = self.queue.get_nowait() if timeout == 0 else self.queue.get(timeout=timeout)
196 self._execute_command(cmd)
197 self.queue.task_done()
198 except queue.Empty:
199 break
200 except Exception as e:
201 print(f"Command processing error: {e}")
202
203 def _execute_command(self, cmd: str) -> None:
204 """Execute a single command with proper error handling."""
205 if self.verbose:
206 print(f"[CMD] Executing: {cmd}")
207
208 try:
209 result = subprocess.run(
210 cmd,
211 shell=True,
212 check=True,
213 stdout=subprocess.PIPE,
214 stderr=subprocess.PIPE,
215 text=True,
216 timeout=30 # Prevent hanging commands
217 )
218
219 if self.verbose:
220 if result.stdout:
221 print(f"[CMD] Output:\n{result.stdout}")
222 if result.stderr:
223 print(f"[CMD] Errors:\n{result.stderr}", file=sys.stderr)
224
225 except subprocess.TimeoutExpired:
226 print(f"[CMD] Timeout: Command took too long: {cmd}")
227 except subprocess.CalledProcessError as e:
228 print(f"[CMD] Failed (exit {e.returncode}): {cmd}\n{e.stderr}")
229 except Exception as e:
230 print(f"[CMD] Unexpected error: {e}")
231
232 def put_command(self, cmd: str, block: bool = True,
233 timeout: Optional[float] = None) -> bool:
234 """
235 Safely add a command to the queue from external sources.
236
237 Returns:
238 bool: True if command was successfully queued
239 """
240 try:
241 with self.lock:
242 self.queue.put(cmd, block=block, timeout=timeout)
243 return True
244 except queue.Full:
245 if self.verbose:
246 print(f"[CMD] Queue full, could not add command: {cmd}")
247 return False

249 def __enter__(self):
250 """Context manager entry point."""
251 self.start_command_listener()

253 self.start_remote_processor()
254 return self

256 def __exit__(self, exc_type, exc_val, exc_tb):
257 """Context manager exit point."""
258 self.stop()

451def set_default_sink_by_name(sink_name: str) -> bool:
452 """Set default sink and move existing sink inputs to it."""
453 if not _HAS_PULSECTL:
454 print("pulsectl not available")
455 return False
456 try:
457 with Pulse('tailsmusic') as p:
458 sinks = {s.name: s for s in p.sink_list()}
459 if sink_name not in sinks:
460 print("Sink not found:", sink_name)
461 return False
462 p.sink_default_set(sinks[sink_name])
463 # Move any active streams to the sink
464 for si in p.sink_input_list():
465 try:
466 p.sink_input_move(si.index, sinks[sink_name].index)
467 except Exception:
468 pass
469 return True
470 except Exception as e:
471 print("Error setting default sink:", e)
472 return False

475def bluetooth_list_devices():
476 """Return list of known bluetooth devices using bluetoothctl."""
477 try:
478 out = subprocess.check_output(['bluetoothctl', 'devices'], text=True)
479 devices = []
480 for line in out.splitlines():
481 # format: Device XX:XX:XX:XX:XX:XX Name
482 parts = line.split(' ', 2)
483 if len(parts) >= 3 and parts[0] == 'Device':
484 mac = parts[1]
485 name = parts[2]
486 devices.append((mac, name))
487 return devices
488 except Exception as e:
489 print('bluetoothctl devices error:', e)
490 return []

493def bluetooth_scan(timeout=5):
494 """Scan for bluetooth devices for `timeout` seconds and return devices."""
495 # Prefer BLE scan with bleak if available
496 if _HAS_BLEAK:
497 try:
498 devices = []
499 results = BleakScanner.discover(timeout=timeout)
500 for d in results:
501 name = d.name or d.metadata.get('local_name') if getattr(d, 'metadata', None) else d.name
502 devices.append((d.address, name or ''))
503 if devices:
504 return devices
505 except Exception as e:
506 print('bleak scan error:', e)
507
508 # Fallback to bluetoothctl interactive scan for non-BLE or if bleak is unavailable
509 try:
510 p = subprocess.Popen(['bluetoothctl'], stdin=subprocess.PIPE, stdout=subprocess.PIPE, stderr=subprocess.PIPE, text=True)
511 try:
512 p.stdin.write('scan on\n')
513 p.stdin.flush()
514 sleep(timeout)
515 p.stdin.write('scan off\n')
516 p.stdin.write('devices\n')
517 p.stdin.write('exit\n')
518 out, err = p.communicate(timeout=timeout + 5)
519 except subprocess.TimeoutExpired:
520 p.kill()
521 out, err = p.communicate()
522 devices = []
523 for line in out.splitlines():
524 if 'Device' in line:
525 parts = line.split()
526 try:
527 i = parts.index('Device')
528 mac = parts[i+1]
529 name = ' '.join(parts[i+2:]) if len(parts) > i+2 else ''
530 devices.append((mac, name))
531 except Exception:
532 continue
533 if not devices:
534 return bluetooth_list_devices()
535 return devices
536 except Exception as e:
537 print('bluetooth scan error:', e)
538 return []

541def bluetooth_pair_connect(mac: str) -> bool:
542 """Attempt to pair, trust, and connect to a device via bluetoothctl."""
543 try:
544 subprocess.run(['bluetoothctl', 'pair', mac], check=False)
545 subprocess.run(['bluetoothctl', 'trust', mac], check=False)

547 return res.returncode == 0
548 except Exception as e:
549 print('bluetooth connect error:', e)
