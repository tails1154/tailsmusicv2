50def menu_nav(event, selected, options):
51 """
52 Handle menu navigation: back cycles BACK, skip cycles FORWARD, play/pause (okbutton/okbutton2) selects.
53 Returns tuple (selected, selected_action) where selected_action is True if play/pause was pressed.
54 """
55 key_event = categorize(event)
56 if key_event.keystate == 1:
57 key = key_event.keycode
58 if key == config['backbutton']:
59 selected = (selected - 1) % len(options)
60 speak(options[selected])
61 elif key == config['skipbutton']:
62 selected = (selected + 1) % len(options)
63 speak(options[selected])
64 elif key in [config['okbutton'], config['okbutton2']]:
65 return selected, True
66 return selected, False

126 def _remote_command_processor(self) -> None:
127 """Process commands in a separate process."""
128 while True:
129 try:
130 cmd = self.queue.get(timeout=1)
131 self._execute_command(cmd)
132 except queue.Empty:
133 if not self.running:
134 break
135 except Exception as e:
136 print(f"Remote processor error: {e}")

138 def stop(self) -> None:
139 """Stop all components of the command queue."""
140 with self.lock:
141 self.running = False
142 if self.listener_thread:
143 self.listener_thread.join(timeout=1.0)
144 if self.process:
145 self.process.join(timeout=1.0)
146 if self.enable_multiprocessing:
147 self.manager.shutdown()

162 def process_Command(self, timeout: float = 0.01) -> None:
163 """
164 Process pending commands in the queue (in current process).
165
166 Args:
167 timeout: Maximum time to wait for a command (seconds)
168 """
169 while True:
170 try:
171 cmd = self.queue.get_nowait() if timeout == 0 else self.queue.get(timeout=timeout)
172 self._execute_command(cmd)
173 self.queue.task_done()
174 except queue.Empty:
175 break
176 except Exception as e:
177 print(f"Command processing error: {e}")

179 def _execute_command(self, cmd: str) -> None:
180 """Execute a single command with proper error handling."""
181 if self.verbose:
182 print(f"[CMD] Executing: {cmd}")
183
184 try:
185 result = subprocess.run(
186 cmd,
187 shell=True,
188 check=True,
189 stdout=subprocess.PIPE,
190 stderr=subprocess.PIPE,
191 text=True,
192 timeout=30 # Prevent hanging commands
193 )
194
195 if self.verbose:
196 if result.stdout:
197 print(f"[CMD] Output:\n{result.stdout}")
198 if result.stderr:
199 print(f"[CMD] Errors:\n{result.stderr}", file=sys.stderr)
200
201 except subprocess.TimeoutExpired:
202 print(f"[CMD] Timeout: Command took too long: {cmd}")
203 except subprocess.CalledProcessError as e:
204 print(f"[CMD] Failed (exit {e.returncode}): {cmd}\n{e.stderr}")
205 except Exception as e:
206 print(f"[CMD] Unexpected error: {e}")

208 def put_command(self, cmd: str, block: bool = True,
209 timeout: Optional[float] = None) -> bool:
210 """
211 Safely add a command to the queue from external sources.
212
213 Returns:
214 bool: True if command was successfully queued
215 """
216 try:
217 with self.lock:
218 self.queue.put(cmd, block=block, timeout=timeout)
219 return True
220 except queue.Full:
221 if self.verbose:
222 print(f"[CMD] Queue full, could not add command: {cmd}")
223 return False
